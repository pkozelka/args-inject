28      public static int trapStandardOutputs(StringBuilder stdout, StringBuilder stderr, Callable<Integer> callable) throws Exception {
29          final PrintStream origOut = System.out;
30          final PrintStream origErr = System.err;
31          final ByteArrayOutputStream outBaos = new ByteArrayOutputStream();
32          final ByteArrayOutputStream errBaos = new ByteArrayOutputStream();
33          if (stdout != null) {
34              System.setOut(new PrintStream(outBaos));
35          }
36          if (stderr != null) {
37              System.setErr(new PrintStream(errBaos));
38          }
39          try {
40              return callable.call();
41          } finally {
42              if (stdout != null) {
43                  outBaos.flush();
44                  stdout.append(outBaos.toString().trim());
45                  System.setOut(origOut);
46              }
47              if (stderr != null) {
