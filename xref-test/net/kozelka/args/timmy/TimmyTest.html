57      /**
58       * Runs {@link net.kozelka.args.timmy.Timmy#run} and returns its standard output as a string.
59       * It imitates execution from commandline but actually does not leave JVM.
60       *
61       * @param args arguments to pass
62       * @return trapped System.out
63       * @throws Exception -
64       */
65      private String timmy(String... args) throws Exception {
66  //        ArgsUtils.debugMode = true;
67          // trap stdout to a string
68          System.err.println("timmy " + Arrays.toString(args));
69          final PrintStream origOut = System.out;
70          final ByteArrayOutputStream baos = new ByteArrayOutputStream();
71          final PrintStream stream = new PrintStream(baos);
72          System.setOut(stream);
73          try {
74              Assert.assertEquals(0, Timmy.run(args));
75              baos.flush();
76              final String result = baos.toString().trim();
77              System.err.println("  ==> " + result);
78              return result;
79          } finally {
80              System.setOut(origOut);
81          }
82      }
