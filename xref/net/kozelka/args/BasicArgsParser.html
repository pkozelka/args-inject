18   * Supports one subcommand on the commandline.
19   * The order of global options, subcommand options, subcommand itself and its parameters does not matter - all can be mixed.
20   * Values of options must however directly follow the option name.
21   */

26          this.declarationSetup = declarationSetup;

29      /**
30       * Parses the commandline args.
31       *
32       * @param args the args array to parse
33       * @return initialized command instance
34       * @throws ParseException -
35       */
36      public Callable<Integer> parse(String... args) throws ParseException {
37          final CommandlineDeclaration declaration = declarationSetup.getDeclaration();
38          final LinkedList<String> argsList = new LinkedList<String>(Arrays.asList(args));
39          ArgsUtils.debug("Parsing commandline args: %s", argsList);
40          final List<ParsedOption> parsedOptions = new ArrayList<ParsedOption>();
41          final LinkedList<String> cmdParams = new LinkedList<String>();
42          while (!argsList.isEmpty()) {
43              final String arg = argsList.removeFirst();
44  //            System.out.println("arg = " + arg);
45              final int n = arg.indexOf('=');
46              final String argName;
47              if (n >= 0) {
48                  argName = arg.substring(0, n);
49                  //TODO: also prepend name as separate option argument
50                  // TODO: somehow, support property options like -Dkey=value
51                  final String argValue = arg.substring(n + 1);
52                  argsList.addFirst(argValue);
53              } else {
54                  //TODO: support -x- meaning boolean option with value of false 
55                  argName = arg;
56              }
57              final OptionDeclaration optionDecl;
58              if (arg.equals("-") || arg.equals("--")) {
59                  // allow stdfile value (-) and multi-varargs separator (--)
60                  cmdParams.add(arg);
61                  continue;
62              } else if (arg.startsWith("--")) {
63                  optionDecl = declaration.lookupLongOption(argName);
64                  if (optionDecl == null) {
65                      // little trick to allow stuff like --help or --version behave as command:
66                      final SubCommandDeclaration cmdDecl = declaration.lookupCommand(arg, false);
67                      if (cmdDecl == null) {
68                          throw new ParseException("invalid option: " + arg, 0);
69                      }
70                      cmdParams.add(arg);
71                      continue;
72                  }
73              } else if (arg.startsWith("-")) {
74                  optionDecl = declaration.lookupShortOption(argName);
75                  if (optionDecl == null) {
76                      final Pattern NUMBER_CHARS = Pattern.compile("-\\d+(\\.\\d+)?");
77                      if (!NUMBER_CHARS.matcher(argName).matches()) {
78                          throw new ParseException("invalid option: " + arg, 0);
79                      }
80                      // if digits (with dot) follow, pass it as a value => this enables support for negative subcommand param values
81                      cmdParams.add(arg);
82                      continue;
83                  }
84              } else {
85                  cmdParams.add(arg);
86                  continue;
87              }
88  //            System.out.println("optionMethod = " + optionMethod);
89              // pre-parse option - command may be still unknown
90              final ParsedOption option = new ParsedOption(argName, optionDecl);
91              option.parse(argsList);
92              parsedOptions.add(option);
93          }
94          ArgsUtils.debug("  cmdParams:     %s", cmdParams);
95          ArgsUtils.debug("  parsedOptions: %s", parsedOptions);
96          // parse command
97          final SubCommandDeclaration defaultSubCommand = declaration.getDefaultSubCommand();
98          SubCommandDeclaration cmdDecl;
99          String cmdName;
100         if (cmdParams.isEmpty()) {
101             if (defaultSubCommand == null) {
102                 throw new ParseException("no subcommand specified", 0);
103             }
104             cmdDecl = defaultSubCommand;
105             cmdName = cmdname(cmdDecl);
106         } else {
107             cmdName = cmdParams.removeFirst();
108             cmdDecl = declaration.lookupCommand(cmdName, false);
109             if (cmdDecl == null) {
110                 if (defaultSubCommand == null) {
111                     throw new ParseException("unknown subcommand: " + cmdName, 0);
112                 }
113                 cmdParams.addFirst(cmdName); // return it back, it's value
114                 cmdDecl = defaultSubCommand;
115                 cmdName = cmdname(cmdDecl);
116             }
117         }
118         final ArgsCommand commandInstance = declarationSetup.createSubCommand(cmdName, cmdDecl, cmdParams);
119         // fail if any token remains
120         if (!cmdParams.isEmpty()) {
121             throw new ParseException("unparsed tokens: " + cmdParams, 0);
122         }
123         declarationSetup.injectOptions(commandInstance, parsedOptions, cmdName);
124         return commandInstance;
125     }

127     private static String cmdname(SubCommandDeclaration cmdDecl) {
128         String name = cmdDecl.getName();
129         return name == null ? "<UNNAMED>" : name;
130     }

132     public static Callable<Integer> parse(ArgsSetup declarationSetup, String... args) throws ParseException {
133         return new BasicArgsParser(declarationSetup).parse(args);
134     }

136     /**
137      * <p>High-level entry point for commandline processing. Builds command according to setup and args, calls it,
138      * and handles any exceptions in a way expectable from commandline application.</p>
139      * <p>Expected usage:</p>
140      * <pre>
141      * ...
142      * public static void main(String[] args) {
143      *   // ... create your commandline setup here ...
144      *   if (!<b> SingleCommandBuilder.main</b>(setup, args)) {
145      *     {@link System#exit(int) System.exit(1)}; // indicate failure to shell
146      *   }
147      * }
148      * ...
149      * </pre>
150      * <p>Note that this way, the dirty job of calling {@link System#exit(int)} is intentionally left up to the caller.</p>
151      *
152      * @param setup commandline declaration
153      * @param args  actual arguments to be processed
154      * @return exit code. It is a good idea to indicate failure to shell by terminating with {@link System#exit(int) System.exit(1)}
155      * @throws Exception used only if {@link net.kozelka.args.ArgsUtils#debugMode}==true
156      */
157     public static int process(ArgsSetup setup, String... args) throws Exception {
158         try {
159             XmlModelSerializer.checkArgsXml(setup.getDeclaration());
160             final int exitCode = new BasicArgsParser(setup).parse(args).call();
161             ArgsUtils.debug("exitCode = %d", exitCode);
162             return exitCode;
163         } catch (ParseException e) {
164             System.err.println("ERROR: " + e.getMessage());
165             if (ArgsUtils.debugMode) {
166                 e.printStackTrace();
167             }
168             return 1;
169         } catch (IllegalArgumentException e) {
170             System.err.println("ERROR: " + e.getClass().getSimpleName() + " - " + e.getMessage());
171             if (ArgsUtils.debugMode) {
172                 e.printStackTrace();
173             }
174             return 1;
175         }
176     }
