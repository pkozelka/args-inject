58      private void introspectOptions(Class<?> classWithOptions, SubCommandDeclaration attachToSubCommandDeclaration) throws ParseException {
59          final String cn = classWithOptions.getName();
60          if (cn.startsWith("java.")) return;
61  //        System.out.println("cls " + cn);
62          for (Class iface : classWithOptions.getInterfaces()) {
63              introspectOptions(iface, attachToSubCommandDeclaration);
64          }
65          final Class<?> superclass = classWithOptions.getSuperclass();
66          if (superclass != null && !superclass.equals(Object.class)) {
67              introspectOptions(superclass, attachToSubCommandDeclaration);
68          }
69          for (Method method : classWithOptions.getMethods()) {
70              if (method.getAnnotation(Option.class) != null) {
71                  final OptionDeclaration optionDeclaration = createOptionDecl(method);
72                  cliDeclaration.addOption(optionDeclaration);
73                  if (attachToSubCommandDeclaration != null) {
74                      attachToSubCommandDeclaration.addOptionDeclaration(optionDeclaration);
75                  }
76              }
77          }
78      }

80      private SubCommandDeclaration createCmdDecl(Class<? extends ArgsCommand> cmdClass) {
81          final SubCommand annCommand = cmdClass.getAnnotation(SubCommand.class);
82          // note: @SubCommand is optional
83          final SubCommandDeclaration cmdDecl = new SubCommandDeclaration(cmdClass);
84          if (annCommand != null) {
85              cmdDecl.setName(annCommand.name().equals("") ? null : annCommand.name());
86              cmdDecl.addAlternateNames(Arrays.asList(annCommand.aliases()));
87              final String desc = annCommand.description();
88              cmdDecl.setDescription("".equals(desc) ? null : desc);
89          }
90          final Constructor<? extends Callable<Integer>> con = findPublicConstructor(cmdDecl.getCommandClass());
91          ParamDeclaration paramDecl = null;
92          for (int i = 0; i < con.getParameterTypes().length; i++) {
93              final Class<?> paramType = con.getParameterTypes()[i];
94              final Annotation[] pa = con.getParameterAnnotations()[i];
95              paramDecl = createParamDecl(paramType, pa);
96              cmdDecl.addParamDeclaration(paramDecl);
97          }
98          // if the constructor is varargs, note it in the last param declaration - which is expected to be array
99          if (paramDecl != null && con.isVarArgs()) {
100             paramDecl.setVarArgs(true);
101         }
102         return cmdDecl;
103     }

105     private OptionDeclaration createOptionDecl(Method method) throws ParseException {
106         final Option annOption = method.getAnnotation(Option.class);
107         final Global annGlobal = method.getAnnotation(Global.class);
108         // note: @Option is mandatory
109         final OptionDeclaration optionDeclaration = new OptionDeclaration(annOption.shortName(), annOption.longName(), method);
110         for (int i = 0; i < method.getParameterTypes().length; i++) {
111             final Class<?> paramType = method.getParameterTypes()[i];
112             final Annotation[] pa = method.getParameterAnnotations()[i];
113             optionDeclaration.addParamDeclaration(createParamDecl(paramType, pa));
114         }
115         final String desc = annOption.description();
116         optionDeclaration.setDescription("".equals(desc) ? null : desc);
117         if (annGlobal != null) {
118             optionDeclaration.setGlobal(annGlobal.value());
119         } else {
120             final boolean isClassAbstract = Modifier.isAbstract(method.getDeclaringClass().getModifiers());
121             final boolean isMethodAbstract = Modifier.isAbstract(method.getModifiers());
122             optionDeclaration.setGlobal(isClassAbstract || isMethodAbstract);
123         }
124         return optionDeclaration;
125     }

127     private ParamDeclaration createParamDecl(Class<?> paramType, Annotation[] pa) {
128         Param annParam = null;
129         for (Annotation annotation : pa) {
130             if (annotation instanceof Param) {
131                 annParam = (Param) annotation;
132             }
133         }
134         // note: @Param is optional
135         final ParamDeclaration paramDecl = new ParamDeclaration(paramType);
136         if (annParam != null) {
137             final String format = annParam.format();
138             paramDecl.setFormat("".equals(format) ? null : format);
139             final String name = annParam.value();
140             paramDecl.setSymbolicName("".equals(name) ? null : name);
141         }
142         if (paramType.isArray()) {
143             final String listSeparator = annParam == null ? "" : annParam.listSeparator();
144             if (!"".equals(listSeparator)) {
145                 paramDecl.setListSeparator(listSeparator);
146             } else if (paramType.getComponentType().equals(File.class)) {
147                 paramDecl.setListSeparator(File.pathSeparator);
148             } else {
149                 paramDecl.setListSeparator(",");
150             }
151         }
152         return paramDecl;
153     }

159     public ArgsCommand createSubCommand(String cmdName, SubCommandDeclaration cmdDecl, LinkedList<String> cmdParams) throws ParseException {
160         final List<Object> unmarshalledValues = ParamDeclaration.parseParamList("subcommand " + cmdName, cmdDecl.getParamDeclarations(), cmdParams);
161         // find public constructor
162         final Class<? extends ArgsCommand> cmdClass = cmdDecl.getCommandClass();
163         final Constructor<? extends ArgsCommand> con = findPublicConstructor(cmdClass);
164         ArgsUtils.debug("cmd constructor: %s%s", con, unmarshalledValues);
165         try {
166             return con.newInstance(unmarshalledValues.toArray(new Object[unmarshalledValues.size()]));
167         } catch (InvocationTargetException e) {
168             throw new IllegalStateException(e.getCause());
169         } catch (IllegalAccessException e) {
170             throw new IllegalStateException(e);
171         } catch (InstantiationException e) {
172             throw new IllegalStateException(e);
173         }
174     }

176     public void injectOptions(ArgsCommand commandInstance, List<ParsedOption> parsedOptions, String cmdName) throws ParseException {
177         final Class<? extends ArgsCommand> cmdClass = commandInstance.getClass();
178         // set options on command instance
179         for (ParsedOption option : parsedOptions) {
180             final List<Object> unmarshalledValues = option.getValues();
181             final Method method = option.getOptionDecl().getOptionMethod();
182             ArgsUtils.debug("  option method: %s%s", method, unmarshalledValues.isEmpty() ? "" : unmarshalledValues);
183             try {
184                 final Object[] values = unmarshalledValues.toArray(new Object[unmarshalledValues.size()]);
185                 final Class<?> declaringClass = method.getDeclaringClass();
186                 if (declaringClass.isAssignableFrom(cmdClass)) {
187                     method.invoke(commandInstance, values);
188                 } else {
189                     throw new ParseException(String.format("subcommand %s does not accept option '%s'",
190                             cmdName, option.getUsedName()), 0);
191                 }
192             } catch (IllegalAccessException e) {
193                 throw new IllegalStateException(e);
194             } catch (InvocationTargetException e) {
195                 throw new IllegalStateException(e);
196             }
197         }
198         // special cases - todo: generalize ?
199         if (commandInstance instanceof MetaCommand) {
200             final MetaCommand h = (MetaCommand) commandInstance;
201             h.setDeclaration(cliDeclaration);
202         }
203     }

206     public static Constructor<ArgsCommand> findPublicConstructor(Class<? extends Callable<Integer>> cmdClass) {
207         //TODO: fail if more public constructors exist
208         for (Constructor<?> constructor : cmdClass.getConstructors()) {
209             if (Modifier.isPublic(constructor.getModifiers())) return (Constructor<ArgsCommand>) constructor;
210         }
211         throw new IllegalStateException("There is no public constructor on " + cmdClass);
