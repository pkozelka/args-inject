23  @SubCommand(name = "help", aliases = {"--help", "?", "h"}, description = "shows help for the whole app. or for specified command")

37      public Integer call() throws Exception {
38          if (command == null) {
39              globalHelp();
40              return 0;
41          } else {
42              final SubCommandDeclaration cmdDecl = declaration.lookupCommand(command, false);
43              if (cmdDecl == null) {
44                  throw new ParseException(command + ": unknown subcommand - cannot help", 0);
45              }
46              commandHelp(cmdDecl);
47              return 0;
48          }
49      }

51      private void commandHelp(SubCommandDeclaration cmdDecl) {
52          final String programName = declaration.getProgramName();
53          final String shortDesc = cmdDecl.getDescription();
54          out.println("Subcommand '" + cmdDecl + "': " + shortDesc);
55          out.println();
56          out.println("Usage:");
57          final StringBuilder usage = new StringBuilder("   ");
58          usage.append(programName);
59          usage.append(" ");
60          usage.append(cmdDecl.getName());
61          usage.append(" [options]");
62          usage.append(paramSynopsis(cmdDecl.getParamDeclarations()));
63          out.println(usage);
64  
65          showValidOptions(cmdDecl.getOptionDeclarations(), false);
66          showValidOptions(cmdDecl.getOptionDeclarations(), true);
67      }

69      private void showValidOptions(Collection<OptionDeclaration> options, boolean global) {
70          out.println();
71          final List<OptionDeclaration> sortedOptions = new ArrayList<OptionDeclaration>(options);
72          if (!sortedOptions.isEmpty()) {
73              Collections.sort(sortedOptions, new Comparator<OptionDeclaration>() {
74                  public int compare(OptionDeclaration o1, OptionDeclaration o2) {
75                      return o1.toString().compareTo(o2.toString());
76                  }
77              });
78              final Map<String, String> optMap = new LinkedHashMap<String, String>();
79              int max = 0;
80              // gather option table, and find longest declaration
81              for (OptionDeclaration optionDeclaration : sortedOptions) {
82                  if (optionDeclaration.isGlobal() != global) continue;
83                  final String strDecl = optionDeclaration.toString() + paramSynopsis(optionDeclaration.getParamDeclarations());
84                  if (strDecl.length() > max) {
85                      max = strDecl.length();
86                  }
87                  final String desc = optionDeclaration.getDescription();
88                  optMap.put(strDecl, desc);
89              }
90              if (!optMap.isEmpty()) {
91                  out.println(global ? "Global options:" : "Valid options:");
92                  // print it
93                  max += 2; // two more spaces
94                  final String fmt = "%-" + max + "s";
95                  printTable(optMap, fmt);
96              }
97          }
98  
99          //todo: add verbose description - from .txt resource located next to .class
100     }

102     private void printTable(Map<String, String> optMap, String fmt) {
103         for (Map.Entry<String, String> entry : optMap.entrySet()) {
104             out.print("   ");
105             out.print(String.format(fmt, entry.getKey()));
106             if (entry.getValue() != null) {
107                 out.print(entry.getValue());
108             }
109             out.println();
110         }
111     }

113     private static String paramSynopsis(List<ParamDeclaration> paramDeclarations) {
114         final StringBuilder sb = new StringBuilder();
115         for (ParamDeclaration paramDecl : paramDeclarations) {
116             final String symbolicName = paramDecl.getSymbolicName();
117             sb.append(" <");
118             sb.append(symbolicName);
119             sb.append(">");
120         }
121         return sb.toString();
122     }

143                 }
144             });
145             final Map<String, String> optMap = new LinkedHashMap<String, String>();
146             int max = 0;
147             // gather option table, and find longest declaration
148             for (SubCommandDeclaration subCommandDeclaration : subcommands) {
149                 final String cmdName = subCommandDeclaration.getName();
150                 if (cmdName.length() > max) {
151                     max = cmdName.length();
152                 }
153                 final String desc = subCommandDeclaration.getDescription();
154                 optMap.put(cmdName, desc);
155             }
156             // print it
157             max += 4; // two more spaces
158             final String fmt = "%-" + max + "s";
159             printTable(optMap, fmt);
160         }
161         final SubCommandDeclaration defaultSubCommand = declaration.getDefaultSubCommand();
162         // show help for default command (if unnamed)
163         if (defaultSubCommand == null) {
164             showValidOptions(declaration.getOptionDeclarations(), true);
165             out.println("Type '" + programName + " help <subcommand>' for help on a specific subcommand");
166         } else if (defaultSubCommand.getName() == null) {
167             showValidOptions(defaultSubCommand.getOptionDeclarations(), false);
168             showValidOptions(defaultSubCommand.getOptionDeclarations(), true);
169         } else {
170             showValidOptions(declaration.getOptionDeclarations(), true);
171             out.println("Type '" + programName + " help <subcommand>' for help on a specific subcommand");
172         }
