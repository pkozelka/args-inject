001package net.kozelka.args;

003import java.text.ParseException;
004import java.util.concurrent.Callable;
005import net.kozelka.args.annotation.AnnottationAwareSetup;
006import org.junit.Assert;
007import org.junit.Ignore;
008import org.junit.Test;
009
010public class ArgsParserTest {
011    @Test
012    public void testMultiFiles() throws Exception {
013        // START SNIPPET: example1
014        final int exitCode = DemoFileLister.run("-D", "a", "b",
015                "false", "17", "/tmp", "/var", "/root",
016                "-C=true",
017                "--property", "xxx", "yyy");
018        // END SNIPPET: example1
019        Assert.assertEquals(0, exitCode);
020    }
021
022    @Test
023    public void testSampleOptionShort() throws Exception {
024        final StringBuilder stdout = new StringBuilder();
025        final StringBuilder stderr = new StringBuilder();
026        final int exitCode = ArgsTestUtils.trapStandardOutputs(stdout, stderr, new Callable<Integer>() {
027            public Integer call() throws Exception {
028                return SampleOptionPrinter.run(
029                        "-i", "42",
030                        "-b", "false",
031                        "-s", "Hello");
032            }
033        });
034        System.out.println(stdout);
035        System.err.println(stderr);
036        Assert.assertEquals(0, exitCode);
037        Assert.assertEquals("42:false:Hello", stdout.toString());
038    }
039
040    @Ignore("TODO - attached options feature is not yet working")
041    @Test
042    public void testSampleOptionSyntaxMix() throws Exception {
043        final StringBuilder stdout = new StringBuilder();

046            public Integer call() throws Exception {
047                return SampleOptionPrinter.run(

058
059    @Test
060    public void testGlobalHelp() throws Exception {
061        // START SNIPPET: default-help
062        final AnnottationAwareSetup setup = new AnnottationAwareSetup("DemoFileLister");
063        setup.addSubCommand(DefaultHelpCommand.class);
064        // END SNIPPET: default-help
065        setup.setDefaultSubCommand(DemoFileLister.class);
066        System.err.println("=== using 'help' ===");
067        BasicArgsParser.parse(setup, "help").call();
068        System.err.println("=== using '--help' ===");
069        BasicArgsParser.parse(setup, "--help").call();
070    }
071
072    @Test
073    public void testCommandHelp() throws Exception {
074        final AnnottationAwareSetup setup = new AnnottationAwareSetup("testCommandHelp");
075        setup.setSubCommands(DefaultHelpCommand.class);
076        final int exitCode = BasicArgsParser.process(setup, "help", "help");
077        Assert.assertEquals(0, exitCode);
078    }
079
080    @Test
081    public void testInvalidCommandHelp() throws Exception {
082        final AnnottationAwareSetup setup = new AnnottationAwareSetup("testCommandHelp");
083        setup.setSubCommands(DefaultHelpCommand.class);
084        final int exitCode = BasicArgsParser.process(setup, "help", "changeEarthRotation");
085        Assert.assertEquals(1, exitCode);
086    }
087
088    @Test(expected = ParseException.class)
089    public void testInvalidCommand() throws Exception {
090        final AnnottationAwareSetup setup = new AnnottationAwareSetup("testInvalidCommand");
091        setup.setSubCommands(DefaultHelpCommand.class);
092        BasicArgsParser.parse(setup, "invaLID").call();
093    }
094
095    // START SNIPPET: sample-main
096    public static void main(String[] args) throws Exception {
097        final AnnottationAwareSetup setup = new AnnottationAwareSetup("myapp");
098        setup.setDefaultSubCommand(DemoFileLister.class);
099        final int exitCode = BasicArgsParser.process(setup, args);
100        if (exitCode != 0) {
101            System.exit(exitCode); // indicate failure to shell
102        }
103    }
104    // END SNIPPET: sample-main
105}
