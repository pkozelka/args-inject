031    @Test
032    public void testDedup() throws Exception {
033        final File a1 = new File("/a/1").getAbsoluteFile();
034        final File a2 = new File("/a/2").getAbsoluteFile();
035        final File a3 = new File("/a/3").getAbsoluteFile();
036        final String S = File.pathSeparator;
037        final String stdout = mpt(a1 + S + a2 + S + a1 + S + a3, "--keep-missing-paths", "true");
038        Assert.assertEquals(a1 + S + a2 + S + a3, stdout);
039    }

049    /**
050     * Runs {@link net.kozelka.args.timmy.Timmy#run} and returns its standard output as a string.
051     * It imitates execution from commandline but actually does not leave JVM.
052     *
053     * @param args arguments to pass
054     * @return trapped System.out
055     * @throws Exception -
056     */
057    private String mpt(String... args) throws Exception {
058//        ArgsUtils.debugMode = true;
059        // trap stdout to a string
060        System.err.println("mpt " + Arrays.toString(args));
061        final PrintStream origOut = System.out;
062        final ByteArrayOutputStream baos = new ByteArrayOutputStream();
063        final PrintStream stream = new PrintStream(baos);
064        System.setOut(stream);
065        try {
066            Assert.assertEquals(0, MyPathTool.run(args));
067            baos.flush();
068            final String result = baos.toString().trim();
069            System.err.println("  ==> " + result);
070            return result;
071        } finally {
072            System.setOut(origOut);
073        }
074    }
075
076    /**
077     * Runs {@link net.kozelka.args.timmy.Timmy#run} and returns its standard output as a string.
078     * It imitates execution from commandline but actually does not leave JVM.
079     * Intended for test that expect failing execution.
080     *
081     * @param args arguments to pass
082     * @return trapped System.err
083     * @throws Exception -
084     */
085    private String mptErr(String... args) throws Exception {
086//        ArgsUtils.debugMode = true;
087        // trap stdout to a string
088        System.err.println("mpt " + Arrays.toString(args));
089        final PrintStream origErr = System.err;
090        final ByteArrayOutputStream baos = new ByteArrayOutputStream();
091        final PrintStream stream = new PrintStream(baos);
092        System.setErr(stream);
093        String result = null;
094        try {
095            Assert.assertEquals(1, MyPathTool.run(args));
096            baos.flush();
097            result = baos.toString().trim();
098            return result;
099        } finally {
100            System.setErr(origErr);
101            System.err.println("  ==> " + result);
102        }
103    }
