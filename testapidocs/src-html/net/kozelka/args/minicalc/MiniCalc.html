001package net.kozelka.args.minicalc;
002
003import net.kozelka.args.BasicArgsParser;
004import net.kozelka.args.DefaultHelpCommand;
005import net.kozelka.args.annotation.AnnottationAwareSetup;
006
007/**
008 * Sample application for <a href="http://code.kozelka.net/args-inject">args-inject</a>
009 * @author <a href="mailto:pkozelka@gmail.com">Petr Kozelka</a>
010 */
011public class MiniCalc {
012
013    // START SNIPPET: run
014    /**
015     * Highest-possible level of invocation usable both from {@link #main} and from unit tests.
016     *
017     * @param args commandline arguments
018     * @return true if successful
019     * @throws Exception -
020     */
021    static int run(String... args) throws Exception {
022        final AnnottationAwareSetup setup = new AnnottationAwareSetup("minicalc");
023        setup.addSubCommand(DefaultHelpCommand.class);
024        setup.addSubCommand(Plus.class);
025        setup.addSubCommand(Minus.class);
026        setup.addSubCommand(Logarithm.class);
027        return BasicArgsParser.process(setup, args);
028    }
029    // END SNIPPET: run
030
031    // START SNIPPET: main
032    public static void main(String[] args) throws Exception {
033        final int exitCode = run(args);
034        if (exitCode != 0) {
035            System.exit(exitCode);
036        }
037    }
038    // END SNIPPET: main
039}
