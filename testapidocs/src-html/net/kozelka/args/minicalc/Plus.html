001package net.kozelka.args.minicalc;
002
003import net.kozelka.args.annotation.SubCommand;
004import net.kozelka.args.api.ArgsCommand;
005
006@SubCommand(name = "plus", description = "computes sum of all given numbers")
007public class Plus implements ArgsCommand {
008    private final int[] numbers;
009
010    public Plus(int... numbers) {
011        this.numbers = numbers;
012    }
013
014    public Integer call() throws Exception {
015        int sum = 0;
016        for (int number : numbers) {
017            sum += number;
018        }
019        System.out.println(sum);
020        return 0;
021    }
022}
