001package net.kozelka.args.minicalc;
002
003import net.kozelka.args.annotation.Param;
004import net.kozelka.args.annotation.SubCommand;
005import net.kozelka.args.api.ArgsCommand;
006
007@SubCommand(name = "minus", description = "computes operand1 - operand2")
008public class Minus implements ArgsCommand {
009    private final int x;
010    private final int y;
011
012    public Minus(@Param("operand1") int x, @Param("operand2") int y) {
013        this.x = x;
014        this.y = y;
015    }
016
017    public Integer call() throws Exception {
018        System.out.println(x - y);
019        return 0;
020    }
021}
