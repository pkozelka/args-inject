001package net.kozelka.args;
002
003import java.io.*;
004import java.net.URL;
005import java.util.concurrent.Callable;
006import org.junit.Assert;
007
008public class ArgsTestUtils {
009    public static void filewrite(String filename, String text) throws IOException {
010        final File dir = getTestClassesDir();
011        final File of = new File(dir, filename);
012        final Writer w = new FileWriter(of);
013        System.out.println("Writing to " + of);
014        try {
015            w.write(text);
016            System.out.println(text);
017        } finally {
018            w.close();
019        }
020    }
021
022    public static File getTestClassesDir() {
023        final URL url = ArgsTestUtils.class.getResource("/ref.txt");
024        Assert.assertNotNull("Failed to find resource ref.txt", url);
025        return new File(url.getPath()).getParentFile();
026    }
027
028    public static int trapStandardOutputs(StringBuilder stdout, StringBuilder stderr, Callable<Integer> callable) throws Exception {
029        final PrintStream origOut = System.out;
030        final PrintStream origErr = System.err;
031        final ByteArrayOutputStream outBaos = new ByteArrayOutputStream();
032        final ByteArrayOutputStream errBaos = new ByteArrayOutputStream();
033        if (stdout != null) {
034            System.setOut(new PrintStream(outBaos));
035        }
036        if (stderr != null) {
037            System.setErr(new PrintStream(errBaos));
038        }
039        try {
040            return callable.call();
041        } finally {
042            if (stdout != null) {
043                outBaos.flush();
044                stdout.append(outBaos.toString().trim());
045                System.setOut(origOut);
046            }
047            if (stderr != null) {
048                errBaos.flush();
049                stderr.append(errBaos.toString().trim());
050                System.setErr(origErr);
051            }
052        }
053    }
054
055}
