001package net.kozelka.args;
002
003import org.junit.Test;
004import org.junit.Assert;
005import java.text.ParseException;
006import java.math.BigDecimal;
007import java.util.concurrent.TimeUnit;
008
009public class ArgsUtilsTest {
010    @Test
011    public void testStringToType() throws ParseException {
012        Assert.assertNull("null", ArgsUtils.stringToType(null, Class.class));
013        Assert.assertEquals("char", 'A', ArgsUtils.stringToType("A", char.class));
014        Assert.assertEquals("boolean/false", false, ArgsUtils.stringToType("false", boolean.class));
015        Assert.assertEquals("boolean/true", true, ArgsUtils.stringToType("true", boolean.class));
016        Assert.assertEquals("byte", (byte) 123, ArgsUtils.stringToType("123", byte.class));
017        Assert.assertEquals("short", (short) 12345, ArgsUtils.stringToType("12345", short.class));
018        Assert.assertEquals("int", 1234567890, ArgsUtils.stringToType("1234567890", int.class));
019        Assert.assertEquals("long", 1234567890123456789L, ArgsUtils.stringToType("1234567890123456789", long.class));
020        Assert.assertEquals("float", (float) 3.141592653, ArgsUtils.stringToType("3.141592653", float.class));
021        Assert.assertEquals("double", 3.141592653, ArgsUtils.stringToType("3.141592653", double.class));
022        Assert.assertEquals("BigDecimal", new BigDecimal("3.14159265379"), ArgsUtils.stringToType("3.14159265379", BigDecimal.class));
023        Assert.assertEquals("TimeUnit", TimeUnit.NANOSECONDS, ArgsUtils.stringToType("NANOSECONDS", TimeUnit.class));
024    }
025
026    @Test(expected = ParseException.class)
027    public void testLongStringToChar() throws ParseException {
028        ArgsUtils.stringToType("AB", char.class);
029    }
030
031    @Test(expected = ParseException.class)
032    public void testEmptyStringToChar() throws ParseException {
033        ArgsUtils.stringToType("", char.class);
034    }
035
036    @Test(expected = IllegalStateException.class)
037    public void testNonStaticValueOf() throws ParseException {
038        ArgsUtils.stringToType("Hello", NonStaticValueOf.class);
039    }
040
041    @Test(expected = IllegalStateException.class)
042    public void testStaticBadTypeValueOf() throws ParseException {
043        ArgsUtils.stringToType("Hello", StaticBadTypeValueOf.class);
044    }
045
046    @Test(expected = IllegalStateException.class)
047    public void testNoValueOf() throws ParseException {
048        ArgsUtils.stringToType("Hello", Object.class);
049    }
050
051    @Test(expected = ParseException.class)
052    public void testFailingValueOf() throws ParseException {
053        ArgsUtils.stringToType("Hello", FailingValueOf.class);
054    }
055
056    @Test(expected = ParseException.class)
057    public void testFailingConstructor() throws ParseException {
058        ArgsUtils.stringToType("Hello", FailingConstructor.class);
059    }
060
061    public static class NonStaticValueOf {
062        public NonStaticValueOf valueOf(String s) {
063            throw new UnsupportedOperationException(s);
064        }
065    }
066
067    public static class StaticBadTypeValueOf {
068        public static String valueOf(String s) {
069            throw new UnsupportedOperationException(s);
070        }
071    }
072
073    public static class FailingValueOf {
074        public static FailingValueOf valueOf(String s) {
075            throw new UnsupportedOperationException(s);
076        }
077    }
078
079    public static class FailingConstructor {
080        public FailingConstructor(String s) {
081            throw new UnsupportedOperationException(s);
082        }
083    }
084}
