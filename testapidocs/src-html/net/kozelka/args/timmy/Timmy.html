001package net.kozelka.args.timmy;
002
003import java.text.SimpleDateFormat;
004import java.util.Date;
005import net.kozelka.args.BasicArgsParser;
006import net.kozelka.args.DefaultHelpCommand;
007import net.kozelka.args.annotation.AnnottationAwareSetup;
008import net.kozelka.args.annotation.Option;
009import net.kozelka.args.annotation.Param;
010import net.kozelka.args.annotation.SubCommand;
011import net.kozelka.args.api.ArgsCommand;
012
013/**
014 * This sample code demonstrates working with input arguments of {@link Date} type.
015 * Please note that processing dates is still in experimental state; the exact way of declaring such parameters can be changed in future versions.
016 * This sample should serve as a reference reflecting up-to-date declaration style.
017 */
018public class Timmy {
019
020    @SubCommand(name = "shift", description = "shifts time by given milliseconds")
021    public static class TimeShiftCommand implements ArgsCommand {
022        //TODO: option "unit" for offset
023        private final Date basetime;
024        private final long offset;
025        private SimpleDateFormat outputTimeFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSS");
026
027        public TimeShiftCommand(@Param("basetime") Date basetime, @Param("offset") long offset) {
028            this.basetime = basetime;
029            this.offset = offset;
030        }
031
032        @Option(shortName = "-f", longName = "--output-time-format", description = "result format spec. (SimpleDateFormat)")
033        public void setOutputTimeFormat(@Param("format") SimpleDateFormat outputTimeFormat) {
034            this.outputTimeFormat = outputTimeFormat;
035        }
036
037        public Integer call() throws Exception {
038            final String result = outputTimeFormat.format(new Date(basetime.getTime() + offset));
039            System.out.println(result);
040            return 0;
041        }
042    }
043
044    @SubCommand(name = "delta", description = "computes difference in millis between two given times")
045    public static class TimeDeltaCommand implements ArgsCommand {
046        //TODO: option "unit" for output (output as decimal)
047        private final Date basetime;
048        private final Date otherTime;
049
050        public TimeDeltaCommand(Date basetime, Date otherTime) {
051            this.basetime = basetime;
052            this.otherTime = otherTime;
053        }
054
055        public Integer call() throws Exception {
056            final long delta = otherTime.getTime() - basetime.getTime();
057            System.out.println(delta);
058            return 0;
059        }
060    }
061
062    @SubCommand(name = "timeconvert", description = "converts given times to millis since epoch; uses current time if none specified")
063    public static class TimeConvertCommand implements ArgsCommand {
064        private final Date[] times;
065
066        public TimeConvertCommand(@Param("times") Date... times) {
067            this.times = times.length == 0 ? new Date[]{new Date()} : times;
068        }
069
070        public Integer call() throws Exception {
071            for (Date time : times) {
072                System.out.println(time.getTime());
073            }
074            return 0;
075        }
076    }
077
078    /**
079     * Highest-possible level of invocation usable both from {@link #main} and from unit tests.
080     *
081     * @param args commandline arguments
082     * @return true if successful
083     * @throws Exception -
084     */
085    static int run(String... args) throws Exception {
086        final AnnottationAwareSetup setup = new AnnottationAwareSetup("timmy");
087        setup.setDefaultSubCommand(TimeConvertCommand.class);
088        setup.addSubCommand(DefaultHelpCommand.class);
089        setup.addSubCommand(TimeShiftCommand.class);
090        setup.addSubCommand(TimeDeltaCommand.class);
091//TODO        setup.addSubCommand(LocalToUtcCommand.class);
092//TODO        setup.addSubCommand(UtcToLocalCommand.class);
093        return BasicArgsParser.process(setup, args);
094    }
095
096    public static void main(String[] args) throws Exception {
097        final int exitCode = run(args);
098        if (exitCode != 0) {
099            System.exit(exitCode);
100        }
101    }
102}
