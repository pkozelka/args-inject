057    /**
058     * Runs {@link net.kozelka.args.timmy.Timmy#run} and returns its standard output as a string.
059     * It imitates execution from commandline but actually does not leave JVM.
060     *
061     * @param args arguments to pass
062     * @return trapped System.out
063     * @throws Exception -
064     */
065    private String timmy(String... args) throws Exception {
066//        ArgsUtils.debugMode = true;
067        // trap stdout to a string
068        System.err.println("timmy " + Arrays.toString(args));
069        final PrintStream origOut = System.out;
070        final ByteArrayOutputStream baos = new ByteArrayOutputStream();
071        final PrintStream stream = new PrintStream(baos);
072        System.setOut(stream);
073        try {
074            Assert.assertEquals(0, Timmy.run(args));
075            baos.flush();
076            final String result = baos.toString().trim();
077            System.err.println("  ==> " + result);
078            return result;
079        } finally {
080            System.setOut(origOut);
081        }
082    }
083
084    /**
085     * Runs {@link net.kozelka.args.timmy.Timmy#run} and returns its standard output as a string.
086     * It imitates execution from commandline but actually does not leave JVM.
087     * Intended for test that expect failing execution.
088     *
089     * @param args arguments to pass
090     * @return trapped System.err
091     * @throws Exception -
092     */
093    private String timmyErr(String... args) throws Exception {
094//        ArgsUtils.debugMode = true;
095        // trap stdout to a string
096        System.err.println("timmy " + Arrays.toString(args));
097        final PrintStream origErr = System.err;
098        final ByteArrayOutputStream baos = new ByteArrayOutputStream();
099        final PrintStream stream = new PrintStream(baos);
100        System.setErr(stream);
101        String result = null;
102        try {
103            Assert.assertEquals(1, Timmy.run(args));
104            baos.flush();
105            result = baos.toString().trim();
106            return result;
107        } finally {
108            System.setErr(origErr);
109            System.err.println("  ==> " + result);
110        }
111    }
