027  /**
028   * Creates new empty document.
029   * @return new document
030   */
031  public static Document createDocument() {
032    try {
033      DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
034      documentBuilderFactory.setNamespaceAware(true);
035      DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
036      return documentBuilder.newDocument();
037    } catch (ParserConfigurationException e) {
038      throw new XmlRuntimeException(e);
039    }
040  }

042  /**
043   * Creates element with specified name and contens.
044   * Type of objects in contents can be one of the following: org.w3c.dom.Attr, org.w3c.dom.Node, java.lang.String and
045   * Iterable or array which can again contain objects with all mentioned types.
046   * @param d document used as factory for creating element
047   * @param name qualified name
048   * @param contents element contents
049   * @return new element with specified name and contens
050   */
051  public static Element createElement(Document d, QName name, Object... contents) {
052    String qualifiedName;
053    if (name.getPrefix().equals(XMLConstants.DEFAULT_NS_PREFIX)) {
054      qualifiedName = name.getLocalPart();
055    } else {
056      qualifiedName = name.getPrefix() + ":" + name.getLocalPart();
057    }
058    Element element = d.createElementNS(name.getNamespaceURI(), qualifiedName);
059    insertBefore(d, element, null, contents);
060    return element;
061  }
062
063  /**
064   * Creates attribute with specified name and value.
065   * @param d document used as factory for creating attribute
066   * @param name qualified name
067   * @param value attribute value
068   * @return new attribute with specified qualified name and value
069   */
070  public static Attr createAttribute(Document d, QName name, String value) {
071    String qualifiedName;
072    if (name.getPrefix().equals(XMLConstants.DEFAULT_NS_PREFIX)) {
073      qualifiedName = name.getLocalPart();
074    } else {
075      qualifiedName = name.getPrefix() + ":" + name.getLocalPart();
076    }
077    Attr attr;
078    attr = d.createAttributeNS(name.getNamespaceURI(), qualifiedName);
079    attr.setValue(value);
080    return attr;
081  }
082
083  private static void insertBefore(Document d, Element parent, Node refChild, Object... contents) {
084    for (Object object : contents) {
085      if (object == null) {
086        continue;
087      } else if (object instanceof Attr) {
088        Attr attr = (Attr)object;
089        if (attr.getNamespaceURI() == null) {
090          if (parent.hasAttribute(attr.getName())) {
091            throw new XmlRuntimeException(String.format("Element %s already contains attribute %s.", getName(parent), getName(attr)));
092          } else {
093            parent.setAttributeNode(attr);
094          }
095        } else {
096          if (parent.hasAttributeNS(attr.getNamespaceURI(), attr.getLocalName())) {
097            throw new XmlRuntimeException(String.format("Element %s already contains attribute %s.", getName(parent), getName(attr)));
098          } else {
099            parent.setAttributeNodeNS(attr);
100          }
101        }
102        parent.setAttributeNodeNS((Attr)object);
103      } else if (object instanceof Node) {
104        parent.insertBefore((Node)object, refChild);
105      } else if (object instanceof Iterable<?>) {
106        for (Object object2 : (Iterable<?>)object) {
107          insertBefore(d, parent, refChild, object2);
108        }
109      } else if (object instanceof Object[]) {
110        insertBefore(d, parent, refChild, (Object[])object);
111      } else if (object instanceof String) {
112        parent.insertBefore(d.createTextNode((String)object), refChild);
113      } else {
114        throw new XmlRuntimeException("Unsupported object type: " + object.getClass().getName());
115      }
116    }
117  }

121  /**
122   * Returns attribute with specified name. If such attribute doesn't exist returns null.
123   * @param parent queried Element
124   * @param name qualified name
125   * @return attribute with specified name. If such attribute doesn't exist returns null
126   */
127  public static Attr getAttribute(final Element parent, final QName name) {
128    if (name.getNamespaceURI() == null || name.getNamespaceURI().equals(XMLConstants.NULL_NS_URI)) {
129      return parent.getAttributeNode(name.getLocalPart());
130    } else {
131      return parent.getAttributeNodeNS(name.getNamespaceURI(), name.getLocalPart());
132    }
133  }
134
135  /**
136   * Returns all descendant elements with specified name.
137   * @param parent queried Element
138   * @param name qualified name
139   * @return all descendant elements with specified name.
140   */
141  @SuppressWarnings("unchecked")
142  public static Iterable<Element> getDescendants(final Element parent, final QName name) {
143    return (Iterable)nodeListToList(parent.getElementsByTagNameNS(name.getNamespaceURI(), name.getLocalPart()));
144  }
145
146  /**
147   * Returns qualified name of specified node.
148   * @param node node which name will be returned
149   * @return qualified name of specified node
150   */
151  public static QName getName(Node node) {
152    String namespaceURI = node.getNamespaceURI();
153    if (namespaceURI == null) {
154      return new QName(node.getNodeName());
155    } else {
156      String prefix = node.getPrefix() != null ? node.getPrefix() : XMLConstants.DEFAULT_NS_PREFIX;
157      return new QName(namespaceURI, node.getLocalName(), prefix);
158    }
159  }
160
161  private static List<Node> nodeListToList(final NodeList nodeList) {
162    return new AbstractList<Node>() {
163
164      @Override
165      public Node get(int index) {
166        return nodeList.item(index);
167      }
168
169      @Override
170      public int size() {
171        return nodeList.getLength();
172      }
173    };
174  }

218  private static Document loadXml(InputSource inputSource) throws XmlException {
219    try {
220      DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
221      documentBuilderFactory.setNamespaceAware(true);
222      documentBuilderFactory.setValidating(false);
223      DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
224      // todo
225//      documentBuilder.setEntityResolver(er);
226      documentBuilder.setErrorHandler(new ErrorHandler() {
227        public void warning(SAXParseException exception) throws SAXException {
228          throw exception;
229        }
230        public void error(SAXParseException exception) throws SAXException {
231          throw exception;
232        }
233        public void fatalError(SAXParseException exception) throws SAXException {
234          throw exception;
235        }
236      });
237      return documentBuilder.parse(inputSource);
238    } catch (ParserConfigurationException e) {
239      throw new XmlRuntimeException(e);
240    } catch (SAXException e) {
241      throw new XmlException(e);
242    } catch (IOException e) {
243      throw new XmlException(e);
244    }
245  }

265    try {
266      OutputStreamWriter writer = new OutputStreamWriter(new FileOutputStream(file), "UTF-8");

329  /**
330   * Returns string representation of specified XML document or element.
331   * @param node XML document or element
332   * @return indented XML
333   */
334  public static String toXmlString(Node node) {
335    final StringWriter stringWriter = new StringWriter();
336    try {
337      saveXml(node, stringWriter);
338    } catch (XmlException e) {
339      throw new XmlRuntimeException(e);
340    }
341    return stringWriter.toString();
342  }

344  private static void saveXml(Node node, StreamResult streamResult, XmlSaveOptions options) throws XmlException {
345    try {
346      TransformerFactory factory = TransformerFactory.newInstance();
347      if (options == null || ! options.isDisableFormatting()) {
348        try {
349          // This works only with Writers
350          factory.setAttribute("indent-number", 2);
351        } catch (IllegalArgumentException e) {
352          // ignore
353        }
354      }
355
356      Transformer transformer = factory.newTransformer();
357      transformer.setOutputProperty(OutputKeys.STANDALONE, "yes");
358      if (!(node instanceof Document)) {
359        transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
360      }
361      if (options == null || ! options.isDisableFormatting()) {
362        transformer.setOutputProperty(OutputKeys.INDENT, "yes");
363      }
364
365      DOMSource domSource = new DOMSource(node);
366      
367      transformer.transform(domSource, streamResult);
368      
369    } catch (TransformerConfigurationException e) {
370      throw new XmlRuntimeException(e);
371    } catch (TransformerException e) {
372      throw new XmlException(e);
373    }
374  }
