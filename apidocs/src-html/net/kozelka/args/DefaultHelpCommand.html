001package net.kozelka.args;
002
003import java.io.PrintStream;
004import java.text.ParseException;
005import java.util.*;
006import net.kozelka.args.annotation.SubCommand;
007import net.kozelka.args.api.MetaCommand;
008import net.kozelka.args.model.CommandlineDeclaration;
009import net.kozelka.args.model.OptionDeclaration;
010import net.kozelka.args.model.ParamDeclaration;
011import net.kozelka.args.model.SubCommandDeclaration;
012
013/**
014 * Usecases:
015 * main help - list of commands:
016 * javaapp help
017 * javaapp BADCMD
018 * detailed help - synopsis of one command
019 * javaapp help SOMECMD
020 * javaapp --help SOMECMD
021 * @todo support also: javaapp SOMECMD --help
022 */
023@SubCommand(name = "help", aliases = {"--help", "?", "h"}, description = "shows help for the whole app. or for specified command")
024public class DefaultHelpCommand implements MetaCommand {
025    private CommandlineDeclaration declaration = null;
026    private final String command;
027    private PrintStream out = System.err;
028
029    public DefaultHelpCommand(String... params) {
030        this.command = params.length == 0 ? null : params[0];
031    }
032
033    public void setDeclaration(CommandlineDeclaration declaration) {
034        this.declaration = declaration;
035    }
036
037    public Integer call() throws Exception {
038        if (command == null) {
039            globalHelp();
040            return 0;
041        } else {
042            final SubCommandDeclaration cmdDecl = declaration.lookupCommand(command, false);
043            if (cmdDecl == null) {
044                throw new ParseException(command + ": unknown subcommand - cannot help", 0);
045            }
046            commandHelp(cmdDecl);
047            return 0;
048        }
049    }
050
051    private void commandHelp(SubCommandDeclaration cmdDecl) {
052        final String programName = declaration.getProgramName();
053        final String shortDesc = cmdDecl.getDescription();
054        out.println("Subcommand '" + cmdDecl + "': " + shortDesc);
055        out.println();
056        out.println("Usage:");
057        final StringBuilder usage = new StringBuilder("   ");
058        usage.append(programName);
059        usage.append(" ");
060        usage.append(cmdDecl.getName());
061        usage.append(" [options]");
062        usage.append(paramSynopsis(cmdDecl.getParamDeclarations()));
063        out.println(usage);
064
065        showValidOptions(cmdDecl.getOptionDeclarations(), false);
066        showValidOptions(cmdDecl.getOptionDeclarations(), true);
067    }
068
069    private void showValidOptions(Collection<OptionDeclaration> options, boolean global) {
070        out.println();
071        final List<OptionDeclaration> sortedOptions = new ArrayList<OptionDeclaration>(options);
072        if (!sortedOptions.isEmpty()) {
073            Collections.sort(sortedOptions, new Comparator<OptionDeclaration>() {
074                public int compare(OptionDeclaration o1, OptionDeclaration o2) {
075                    return o1.toString().compareTo(o2.toString());
076                }
077            });
078            final Map<String, String> optMap = new LinkedHashMap<String, String>();
079            int max = 0;
080            // gather option table, and find longest declaration
081            for (OptionDeclaration optionDeclaration : sortedOptions) {
082                if (optionDeclaration.isGlobal() != global) continue;
083                final String strDecl = optionDeclaration.toString() + paramSynopsis(optionDeclaration.getParamDeclarations());
084                if (strDecl.length() > max) {
085                    max = strDecl.length();
086                }
087                final String desc = optionDeclaration.getDescription();
088                optMap.put(strDecl, desc);
089            }
090            if (!optMap.isEmpty()) {
091                out.println(global ? "Global options:" : "Valid options:");
092                // print it
093                max += 2; // two more spaces
094                final String fmt = "%-" + max + "s";
095                printTable(optMap, fmt);
096            }
097        }
098
099        //todo: add verbose description - from .txt resource located next to .class
100    }
101
102    private void printTable(Map<String, String> optMap, String fmt) {
103        for (Map.Entry<String, String> entry : optMap.entrySet()) {
104            out.print("   ");
105            out.print(String.format(fmt, entry.getKey()));
106            if (entry.getValue() != null) {
107                out.print(entry.getValue());
108            }
109            out.println();
110        }
111    }
112
113    private static String paramSynopsis(List<ParamDeclaration> paramDeclarations) {
114        final StringBuilder sb = new StringBuilder();
115        for (ParamDeclaration paramDecl : paramDeclarations) {
116            final String symbolicName = paramDecl.getSymbolicName();
117            sb.append(" <");
118            sb.append(symbolicName);
119            sb.append(">");
120        }
121        return sb.toString();
122    }
123
124    public void globalHelp() {
125        final String programName = declaration.getProgramName();
126        final List<SubCommandDeclaration> subcommands = new ArrayList<SubCommandDeclaration>(declaration.getCommandDeclarations());
127        //TODO: handle commands like help specially
128        out.println("Usage:");
129        final StringBuilder usage = new StringBuilder("   ");
130        usage.append(programName);
131        if (!subcommands.isEmpty()) {
132            usage.append(" <subcommand>");
133        }
134        usage.append(" [options]");
135        usage.append(" [args]");
136        out.println(usage);
137        out.println();
138        if (!subcommands.isEmpty()) {
139            out.println("Available subcommands:");
140            Collections.sort(subcommands, new Comparator<SubCommandDeclaration>() {
141                public int compare(SubCommandDeclaration o1, SubCommandDeclaration o2) {
142                    return o1.getName().compareToIgnoreCase(o2.getName());
143                }
144            });
145            final Map<String, String> optMap = new LinkedHashMap<String, String>();
146            int max = 0;
147            // gather option table, and find longest declaration
148            for (SubCommandDeclaration subCommandDeclaration : subcommands) {
149                final String cmdName = subCommandDeclaration.getName();
150                if (cmdName.length() > max) {
151                    max = cmdName.length();
152                }
153                final String desc = subCommandDeclaration.getDescription();
154                optMap.put(cmdName, desc);
155            }
156            // print it
157            max += 4; // two more spaces
158            final String fmt = "%-" + max + "s";
159            printTable(optMap, fmt);
160        }
161        final SubCommandDeclaration defaultSubCommand = declaration.getDefaultSubCommand();
162        // show help for default command (if unnamed)
163        if (defaultSubCommand == null) {
164            showValidOptions(declaration.getOptionDeclarations(), true);
165            out.println("Type '" + programName + " help <subcommand>' for help on a specific subcommand");
166        } else if (defaultSubCommand.getName() == null) {
167            showValidOptions(defaultSubCommand.getOptionDeclarations(), false);
168            showValidOptions(defaultSubCommand.getOptionDeclarations(), true);
169        } else {
170            showValidOptions(declaration.getOptionDeclarations(), true);
171            out.println("Type '" + programName + " help <subcommand>' for help on a specific subcommand");
172        }
173    }
174
175}
