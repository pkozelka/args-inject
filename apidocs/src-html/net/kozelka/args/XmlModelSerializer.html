001package net.kozelka.args;
002
003import java.io.File;
004import java.lang.reflect.Method;
005import java.util.List;
006import javax.xml.namespace.QName;
007import net.kozelka.args.model.CommandlineDeclaration;
008import net.kozelka.args.model.OptionDeclaration;
009import net.kozelka.args.model.ParamDeclaration;
010import net.kozelka.args.model.SubCommandDeclaration;
011import net.sourceforge.xmlfacade.XmlException;
012import net.sourceforge.xmlfacade.XmlFacade;
013import org.w3c.dom.Document;
014import org.w3c.dom.Element;
015
016public class XmlModelSerializer {
017    public static final String NS = "http://code.kozelka.net/args-inject";
018
019    /**
020     * Creates XML DOM Document containing complete commandline structure declaration
021     *
022     * @param declaration -
023     * @return dom document
024     */
025    public static Document createDomDescriptor(CommandlineDeclaration declaration) {
026        final Document doc = XmlFacade.createDocument();
027        final Element root = XmlFacade.createElement(doc, new QName(NS, "commandline"));
028        root.setAttribute("program", declaration.getProgramName());
029        doc.appendChild(root);
030        if (declaration.getDefaultSubCommand() != null) {
031            createCommandElement(root, declaration.getDefaultSubCommand(), true);
032        }
033        for (SubCommandDeclaration decl : declaration.getCommandDeclarations()) {
034            createCommandElement(root, decl, false);
035        }
036        // all option declarations
037        for (OptionDeclaration decl : declaration.getOptionDeclarations()) {
038            final Method method = decl.getOptionMethod();
039            final Element opt = XmlFacade.createElement(doc, new QName(NS, "option"));
040            root.appendChild(opt);
041            opt.setAttribute("id", method.getDeclaringClass().getName() + ":" + method.getName());
042            createDescriptionElement(opt, decl.getDescription());
043            final String longName = decl.getLongName();
044            final String shortName = decl.getShortName();
045            if (longName != null) {
046                opt.setAttribute("long-name", longName);
047            }
048            if (shortName != null) {
049                opt.setAttribute("short-name", shortName);
050            }
051            if (decl.isGlobal()) {
052                opt.setAttribute("global", "true");
053            }
054            createParamElements(opt, decl.getParamDeclarations());
055        }
056        return doc;
057    }
058
059    static void createCommandElement(Element parent, SubCommandDeclaration decl, boolean isDefault) {
060        final Document doc = parent.getOwnerDocument();
061        final Element cmd = XmlFacade.createElement(doc, new QName(NS, "command"));
062        parent.appendChild(cmd);
063        cmd.setAttribute("name", decl.getName());
064        cmd.setAttribute("class", decl.getCommandClass().getCanonicalName());
065        if (isDefault) {
066            cmd.setAttribute("default", "true");
067        }
068        createDescriptionElement(cmd, decl.getDescription());
069        // aliases
070        for (String alias : decl.getAlternateNames()) {
071            final Element aliasElem = XmlFacade.createElement(doc, new QName(NS, "alias"));
072            cmd.appendChild(aliasElem);
073            aliasElem.setAttribute("name", alias);
074        }
075        createParamElements(cmd, decl.getParamDeclarations());
076
077        // valid options
078        for (OptionDeclaration optionDeclaration : decl.getOptionDeclarations()) {
079            final Method method = optionDeclaration.getOptionMethod();
080            final Element opt = XmlFacade.createElement(doc, new QName(NS, "valid-option"));
081            opt.setAttribute("ref", method.getDeclaringClass().getName() + ":" + method.getName());
082            cmd.appendChild(opt);
083        }
084    }
085
086    static void createDescriptionElement(Element parent, String description) {
087        if (description != null) {
088            final Document doc = parent.getOwnerDocument();
089            parent.appendChild(XmlFacade.createElement(doc, new QName(NS, "description"), doc.createTextNode(description)));
090        }
091    }
092
093    static void createParamElements(Element parent, List<ParamDeclaration> params) {
094        final Document doc = parent.getOwnerDocument();
095        // params
096        for (ParamDeclaration paramDecl : params) {
097            final Element param = XmlFacade.createElement(doc, new QName(NS, "param"));
098            param.setAttribute("symbolic-name", paramDecl.getSymbolicName());
099            param.setAttribute("type", paramDecl.getType().getCanonicalName());
100            parent.appendChild(param);
101            final String format = paramDecl.getFormat();
102            if (format != null) {
103                param.setAttribute("format", format);
104            }
105            final String listSep = paramDecl.getListSeparator();
106            if (paramDecl.isVarArgs()) {
107                param.setAttribute("varargs", "true");
108            } else if (listSep != null) {
109                param.setAttribute("list-separator", listSep);
110            }
111        }
112    }
113
114    /**
115     * If property args.xml is set, its value is used as the name of output file which receives xml representation of the model.
116     *
117     * @param declaration the declaration to print
118     * @throws XmlException -
119     */
120    public static void checkArgsXml(CommandlineDeclaration declaration) throws XmlException {
121        final String xmlFileName = System.getProperty("args.xml");
122        if (xmlFileName != null) {
123            final Document doc = createDomDescriptor(declaration);
124            final File file = new File(xmlFileName);
125            XmlFacade.saveXml(doc, file);
126        }
127    }
128}
