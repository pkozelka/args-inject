001package net.kozelka.args;
002
003import java.text.ParseException;
004import java.util.ArrayList;
005import java.util.Arrays;
006import java.util.LinkedList;
007import java.util.List;
008import java.util.concurrent.Callable;
009import java.util.regex.Pattern;
010import net.kozelka.args.api.ArgsSetup;
011import net.kozelka.args.api.ArgsCommand;
012import net.kozelka.args.model.CommandlineDeclaration;
013import net.kozelka.args.model.OptionDeclaration;
014import net.kozelka.args.model.ParsedOption;
015import net.kozelka.args.model.SubCommandDeclaration;
016
017/**
018 * Supports one subcommand on the commandline.
019 * The order of global options, subcommand options, subcommand itself and its parameters does not matter - all can be mixed.
020 * Values of options must however directly follow the option name.
021 */
022public class BasicArgsParser {
023    private final ArgsSetup declarationSetup;
024
025    public BasicArgsParser(ArgsSetup declarationSetup) {
026        this.declarationSetup = declarationSetup;
027    }
028
029    /**
030     * Parses the commandline args.
031     *
032     * @param args the args array to parse
033     * @return initialized command instance
034     * @throws ParseException -
035     */
036    public Callable<Integer> parse(String... args) throws ParseException {
037        final CommandlineDeclaration declaration = declarationSetup.getDeclaration();
038        final LinkedList<String> argsList = new LinkedList<String>(Arrays.asList(args));
039        ArgsUtils.debug("Parsing commandline args: %s", argsList);
040        final List<ParsedOption> parsedOptions = new ArrayList<ParsedOption>();
041        final LinkedList<String> cmdParams = new LinkedList<String>();
042        while (!argsList.isEmpty()) {
043            final String arg = argsList.removeFirst();
044//            System.out.println("arg = " + arg);
045            final int n = arg.indexOf('=');
046            final String argName;
047            if (n >= 0) {
048                argName = arg.substring(0, n);
049                //TODO: also prepend name as separate option argument
050                // TODO: somehow, support property options like -Dkey=value
051                final String argValue = arg.substring(n + 1);
052                argsList.addFirst(argValue);
053            } else {
054                //TODO: support -x- meaning boolean option with value of false 
055                argName = arg;
056            }
057            final OptionDeclaration optionDecl;
058            if (arg.equals("-") || arg.equals("--")) {
059                // allow stdfile value (-) and multi-varargs separator (--)
060                cmdParams.add(arg);
061                continue;
062            } else if (arg.startsWith("--")) {
063                optionDecl = declaration.lookupLongOption(argName);
064                if (optionDecl == null) {
065                    // little trick to allow stuff like --help or --version behave as command:
066                    final SubCommandDeclaration cmdDecl = declaration.lookupCommand(arg, false);
067                    if (cmdDecl == null) {
068                        throw new ParseException("invalid option: " + arg, 0);
069                    }
070                    cmdParams.add(arg);
071                    continue;
072                }
073            } else if (arg.startsWith("-")) {
074                optionDecl = declaration.lookupShortOption(argName);
075                if (optionDecl == null) {
076                    final Pattern NUMBER_CHARS = Pattern.compile("-\\d+(\\.\\d+)?");
077                    if (!NUMBER_CHARS.matcher(argName).matches()) {
078                        throw new ParseException("invalid option: " + arg, 0);
079                    }
080                    // if digits (with dot) follow, pass it as a value => this enables support for negative subcommand param values
081                    cmdParams.add(arg);
082                    continue;
083                }
084            } else {
085                cmdParams.add(arg);
086                continue;
087            }
088//            System.out.println("optionMethod = " + optionMethod);
089            // pre-parse option - command may be still unknown
090            final ParsedOption option = new ParsedOption(argName, optionDecl);
091            option.parse(argsList);
092            parsedOptions.add(option);
093        }
094        ArgsUtils.debug("  cmdParams:     %s", cmdParams);
095        ArgsUtils.debug("  parsedOptions: %s", parsedOptions);
096        // parse command
097        final SubCommandDeclaration defaultSubCommand = declaration.getDefaultSubCommand();
098        SubCommandDeclaration cmdDecl;
099        String cmdName;
100        if (cmdParams.isEmpty()) {
101            if (defaultSubCommand == null) {
102                throw new ParseException("no subcommand specified", 0);
103            }
104            cmdDecl = defaultSubCommand;
105            cmdName = cmdname(cmdDecl);
106        } else {
107            cmdName = cmdParams.removeFirst();
108            cmdDecl = declaration.lookupCommand(cmdName, false);
109            if (cmdDecl == null) {
110                if (defaultSubCommand == null) {
111                    throw new ParseException("unknown subcommand: " + cmdName, 0);
112                }
113                cmdParams.addFirst(cmdName); // return it back, it's value
114                cmdDecl = defaultSubCommand;
115                cmdName = cmdname(cmdDecl);
116            }
117        }
118        final ArgsCommand commandInstance = declarationSetup.createSubCommand(cmdName, cmdDecl, cmdParams);
119        // fail if any token remains
120        if (!cmdParams.isEmpty()) {
121            throw new ParseException("unparsed tokens: " + cmdParams, 0);
122        }
123        declarationSetup.injectOptions(commandInstance, parsedOptions, cmdName);
124        return commandInstance;
125    }
126
127    private static String cmdname(SubCommandDeclaration cmdDecl) {
128        String name = cmdDecl.getName();
129        return name == null ? "<UNNAMED>" : name;
130    }
131
132    public static Callable<Integer> parse(ArgsSetup declarationSetup, String... args) throws ParseException {
133        return new BasicArgsParser(declarationSetup).parse(args);
134    }
135
136    /**
137     * <p>High-level entry point for commandline processing. Builds command according to setup and args, calls it,
138     * and handles any exceptions in a way expectable from commandline application.</p>
139     * <p>Expected usage:</p>
140     * <pre>
141     * ...
142     * public static void main(String[] args) {
143     *   // ... create your commandline setup here ...
144     *   if (!<b> SingleCommandBuilder.main</b>(setup, args)) {
145     *     {@link System#exit(int) System.exit(1)}; // indicate failure to shell
146     *   }
147     * }
148     * ...
149     * </pre>
150     * <p>Note that this way, the dirty job of calling {@link System#exit(int)} is intentionally left up to the caller.</p>
151     *
152     * @param setup commandline declaration
153     * @param args  actual arguments to be processed
154     * @return exit code. It is a good idea to indicate failure to shell by terminating with {@link System#exit(int) System.exit(1)}
155     * @throws Exception used only if {@link net.kozelka.args.ArgsUtils#debugMode}==true
156     */
157    public static int process(ArgsSetup setup, String... args) throws Exception {
158        try {
159            XmlModelSerializer.checkArgsXml(setup.getDeclaration());
160            final int exitCode = new BasicArgsParser(setup).parse(args).call();
161            ArgsUtils.debug("exitCode = %d", exitCode);
162            return exitCode;
163        } catch (ParseException e) {
164            System.err.println("ERROR: " + e.getMessage());
165            if (ArgsUtils.debugMode) {
166                e.printStackTrace();
167            }
168            return 1;
169        } catch (IllegalArgumentException e) {
170            System.err.println("ERROR: " + e.getClass().getSimpleName() + " - " + e.getMessage());
171            if (ArgsUtils.debugMode) {
172                e.printStackTrace();
173            }
174            return 1;
175        }
176    }
