001package net.kozelka.args.model;
002
003import java.lang.reflect.Method;
004import java.text.ParseException;
005import java.util.HashMap;
006import java.util.HashSet;
007import java.util.Map;
008import java.util.Set;
009
010public final class CommandlineDeclaration {
011    private final Map<String, OptionDeclaration> optionsByShortName = new HashMap<String, OptionDeclaration>();
012    private final Map<String, OptionDeclaration> optionsByLongName = new HashMap<String, OptionDeclaration>();
013
014    private String programName;
015    private SubCommandDeclaration defaultSubCommand;
016    private final Set<SubCommandDeclaration> subCommandDeclarations = new HashSet<SubCommandDeclaration>();
017    private final Set<OptionDeclaration> optionDeclarations = new HashSet<OptionDeclaration>();
018
019    public SubCommandDeclaration getDefaultSubCommand() {
020        return defaultSubCommand;
021    }
022
023    public void setDefaultCommand(SubCommandDeclaration defaultSubCommand) {
024        this.defaultSubCommand = defaultSubCommand;
025    }
026
027    public void addSubCommand(SubCommandDeclaration cmdDecl) {
028        subCommandDeclarations.add(cmdDecl);
029    }
030
031    public Set<OptionDeclaration> getOptionDeclarations() {
032        return optionDeclarations;
033    }
034
035    public void addOption(OptionDeclaration optionDeclaration) throws ParseException {
036        boolean isnew = true;
037        final String shortName = optionDeclaration.getShortName();
038        if (shortName != null) {
039            final OptionDeclaration oldDecl = optionsByLongName.get(shortName);
040            final Method method = optionDeclaration.getOptionMethod();
041            if (oldDecl != null) {
042                if (!oldDecl.getOptionMethod().equals(method)) {
043                    throw new ParseException(String.format("%s: short option name declared twice - first in %s and also in %s",
044                            shortName, oldDecl.getOptionMethod(), method), 0);
045                }
046                isnew = false;
047            }
048            optionsByShortName.put(shortName, optionDeclaration);
049        }
050        final String longName = optionDeclaration.getLongName();
051        if (longName != null) {
052            final OptionDeclaration oldDecl = optionsByLongName.get(longName);
053            final Method method = optionDeclaration.getOptionMethod();
054            if (oldDecl != null) {
055                if (!oldDecl.getOptionMethod().equals(method)) {
056                    throw new ParseException(String.format("%s: long option name declared twice - first in %s and also in %s",
057                            longName, oldDecl.getOptionMethod(), method), 0);
058                }
059                isnew = false;
060            }
061            optionsByLongName.put(longName, optionDeclaration);
062        }
063        if (isnew) {
064            optionDeclarations.add(optionDeclaration);
065        }
066    }
067
068    public SubCommandDeclaration lookupCommand(String command, boolean useDefault) {
069        for (SubCommandDeclaration decl : subCommandDeclarations) {
070            if (command.equals(decl.getName())) {
071                return decl;
072            }
073            if (decl.getAlternateNames().contains(command)) {
074                return decl;
075            }
076        }
077        return useDefault ? defaultSubCommand : null;
078    }
079
080    public OptionDeclaration lookupShortOption(String option) {
081        //TODO: support value with/next to option name
082        return optionsByShortName.get(option);
083    }
084
085    public OptionDeclaration lookupLongOption(String option) {
086        //TODO: support value assignment/next to option name
087        return optionsByLongName.get(option);
088    }
089
090    public Set<SubCommandDeclaration> getCommandDeclarations() {
091        return subCommandDeclarations;
092    }
093
094    public String getProgramName() {
095        return programName;
096    }
097
098    public void setProgramName(String programName) {
099        this.programName = programName;
100    }
101}
