021    private String defaultSymName() {
022        if (isVarArgs()) {
023            return type.getComponentType().getSimpleName().toLowerCase() + "...";
024        } else if (type.isArray()) {
025            return type.getComponentType().getSimpleName().toLowerCase() + "s"; // dirty way to make plural form
026        } else if (type.equals(boolean.class) || type.equals(Boolean.class)) {
027            return "true|false";
028        } else if (Enum.class.isAssignableFrom(type)) {
029            @SuppressWarnings("unchecked")
030            final Class<Enum> etype = (Class<Enum>) type;
031            try {
032                final Object valuesObject = etype.getMethod("values").invoke(null);
033                final Object[] values = (Object[]) valuesObject;
034                final StringBuilder sb = new StringBuilder();
035                for (Object value : values) {
036                    if (sb.length() > 0) {
037                        sb.append('|');
038                    }
039                    sb.append(value);
040                }
041                return sb.toString();
042            } catch (NoSuchMethodException e) {
043                e.printStackTrace();
044            } catch (InvocationTargetException e) {
045                e.printStackTrace();
046            } catch (IllegalAccessException e) {
047                e.printStackTrace();
048            }
049            return "enum";
050        } else {
051            return type.getSimpleName().toLowerCase();
052        }
053    }

087    private Object stringToType(String stringValue, Class<?> paramType) throws ParseException {
088        // TODO: support for custom per-option or per-type unmarshalling should be implemented here
089        // TODO: the value of 'format' field should be thoroughly reconsidered. Shouldn't we support more formats at once ? Should we expose SDF's parameter or wrap it with a logical nama like, DATE, TIME, DATETIME, TIMEMILLIS, DATETIMEMILLIS etc ? This needs to be clarified in light of some real app. usecases
090        if (paramType.equals(Date.class)) {
091            // support Date using format
092            final SimpleDateFormat f = new SimpleDateFormat(format == null ? "yyyy-MM-dd'T'HH:mm:ss.SSS" : format);
093            return f.parse(stringValue);
094        } else if (paramType.equals(Calendar.class)) {
095            // support Calendar using format
096            final Calendar cal = Calendar.getInstance();
097            final SimpleDateFormat f = new SimpleDateFormat(format == null ? "yyyy-MM-dd'T'HH:mm:ss.SSS" : format);
098            final Date date = f.parse(stringValue);
099            cal.setTime(date);
100            return cal;
101        } else {
102            return ArgsUtils.stringToType(stringValue, paramType);
103        }
104    }

106    /**
107     * Parses list of values, be it for option or for command.
108     *
109     * @param context           information to pass to exception message, to be more user friendly
110     * @param paramDeclarations the param declarations
111     * @param actualValues      values to parse against declarations; this list will be shortened according to paramDeclarations
112     * @return list of unmarshalled values
113     * @throws java.text.ParseException -
114     */
115    public static List<Object> parseParamList(String context, List<ParamDeclaration> paramDeclarations, LinkedList<String> actualValues) throws ParseException {
116        // parse remaining tokens as the public constructor's params
117        final List<Object> unmarshalledValues = new ArrayList<Object>();
118        for (ParamDeclaration paramDecl : paramDeclarations) {
119            final Class paramType = paramDecl.getType();
120            if (paramDecl.isVarArgs()) {
121                final List<Object> list = new ArrayList<Object>();
122                while (!actualValues.isEmpty()) {
123                    final String value = actualValues.removeFirst();
124                    if ("--".equals(value)) break; // this allows multiple varargs separated by "--" token
125                    list.add(paramDecl.stringToType(value, paramType.getComponentType()));
126                }
127                final Object array = Array.newInstance(paramType.getComponentType(), list.size());
128                int i = 0;
129                for (Object item : list) {
130                    Array.set(array, i, item);
131                    i++;
132                }
133                unmarshalledValues.add(array);
134//                unmarshalledValues.add(list.toArray((Object[]) array));
135            } else if (actualValues.isEmpty()) {
136                throw new ParseException(context + ": not enough parameters provided", 0);
137            } else if (paramType.isArray()) {
138                final Class componentType = paramType.getComponentType();
139                final String values = actualValues.removeFirst();
140                final List<Object> list = new ArrayList<Object>();
141                for (StringTokenizer tok = new StringTokenizer(values, paramDecl.getListSeparator()); tok.hasMoreTokens();) {
142                    final String value = tok.nextToken();
143                    list.add(paramDecl.stringToType(value, componentType));
144                }
145                final Object array = Array.newInstance(componentType, list.size());
146                unmarshalledValues.add(list.toArray((Object[]) array));
147            } else {
148                final String value = actualValues.removeFirst();
149                unmarshalledValues.add(paramDecl.stringToType(value, paramType));
150            }
151        }
152        return unmarshalledValues;
153    }
