001package net.kozelka.args.model;
002
003import java.lang.reflect.Method;
004import java.util.*;
005import net.kozelka.args.api.ArgsCommand;
006
007public final class SubCommandDeclaration {
008    private String name;
009    private final List<String> alternateNames = new ArrayList<String>();
010    private final Class<? extends ArgsCommand> commandClass;
011    private final List<ParamDeclaration> paramDeclarations = new ArrayList<ParamDeclaration>();
012    private final Map<Method, OptionDeclaration> optionDeclarations = new HashMap<Method, OptionDeclaration>();
013    private String description;
014
015    public SubCommandDeclaration(Class<? extends ArgsCommand> commandClass) {
016        this.commandClass = commandClass;
017    }
018
019    public Class<? extends ArgsCommand> getCommandClass() {
020        return commandClass;
021    }
022
023    public String getName() {
024        return name;
025    }
026
027    public void setName(String name) {
028        this.name = name;
029    }
030
031    public String getDescription() {
032        return description;
033    }
034
035    public void setDescription(String description) {
036        this.description = description;
037    }
038
039    public List<String> getAlternateNames() {
040        return alternateNames;
041    }
042
043    public void addAlternateNames(List<String> alternateNames) {
044        this.alternateNames.addAll(alternateNames);
045    }
046
047    public Set<OptionDeclaration> getOptionDeclarations() {
048        return new HashSet<OptionDeclaration>(optionDeclarations.values());
049    }
050
051    public List<ParamDeclaration> getParamDeclarations() {
052        return paramDeclarations;
053    }
054
055    public void addParamDeclaration(ParamDeclaration paramDeclaration) {
056        paramDeclarations.add(paramDeclaration);
057    }
058
059    public void addOptionDeclaration(OptionDeclaration optionDeclaration) {
060        optionDeclarations.put(optionDeclaration.getOptionMethod(), optionDeclaration);
061    }
062
063    @Override
064    public String toString() {
065        final StringBuilder sb = new StringBuilder();
066        sb.append(name);
067        if (!alternateNames.isEmpty()) {
068            sb.append(" (");
069
070            boolean first = true;
071            for (String altName : alternateNames) {
072                if (first) {
073                    first = false;
074                } else {
075                    sb.append(", ");
076                }
077                sb.append(altName);
078            }
079            sb.append(")");
080        }
081        return sb.toString();
082    }
083}
